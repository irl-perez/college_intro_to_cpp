// functions
char determineGrade(score)
{
    char letter_grade; 
    if score <= 59
        letter_grade = 'F'
    else if score <= 69
        letter_grade = 'D'
    else if score <= 79
        letter_grade = 'C'
    else if score <= 89
        letter_grade = 'B'
    else
        letter_grade = 'A'

    return letter_grade;        
}

double calcAverage(double score1, double score2, double score3, double score4, double score5)
{
    return (score1 + score2 + score3 + score4 + score5) / 5;
}
